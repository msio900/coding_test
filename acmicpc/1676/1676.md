```python
import sys
from functools import reduce
# 10! = 10*9*8*...*1

N = int(sys.stdin.readline())
f = reduce(lambda x, y: x * y, [i for i in range(1, N+1)])

for i in reversed(str(f)):
    if i == '0':
        answer += 1
    elif i != '0':
        break
```
