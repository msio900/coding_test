import sys

input = sys.stdin.readline

N = int(input())

hi = [int(input()) for _ in range(N)]

cnt = 0
stack = []
for i in range(N):
    while stack and stack[-1] <= hi[i]:
        stack.pop()
    stack.append(hi[i])
    cnt += len(stack) - 1

print(cnt)
