    #include <bits/stdc++.h>
    using namespace std;

    const int INF = 0x3f3f3f3f;
    int d[105][105];
    int n, m;

    int main(void){
      ios::sync_with_stdio(0);
      cin.tie(0);

      cin >> n >> m;
      for(int i = 1; i <= n; i++)
        fill(d[i], d[i]+1+n, INF);
      while(m--){
        int a,b,c;
        cin >> a >> b >> c;
        d[a][b] = min(d[a][b], c);
      }
      for(int i = 1; i <= n; i++) d[i][i] = 0;

      for(int k = 1; k <= n; k++)
        for(int i = 1; i <= n; i++)
          for(int j = 1; j <= n; j++)
            d[i][j] = min(d[i][j], d[i][k]+d[k][j]);

      for(int i = 1; i <= n; i++){
        for(int j = 1; j <= n; j++){
          if(d[i][j] == INF) cout << "0 ";
          else cout << d[i][j] << ' ';
        }
        cout << '\n';
      }
    }

      matrix[i][i] = 0
  
  for s in range(n):
      for i in range(n):
          for j in range(n):
              if i != s or j != s:
                  matrix[i][j] = min(matrix[i][s]+matrix[s][j], matrix[i][j])
  
  for i in range(n):
      for j in range(n):
          print(0 if matrix[i][j] == 100000000 else matrix[i][j], end=' ')
