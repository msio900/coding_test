from copy import deepcopy

is_wolf = list()

def find_max_recursive(current_loc, used, nsheep, nwolf, can_go):
    global max_sheep

    if used[current_loc]: return  # ν„μ¬ λ…Έλ“λ¥Ό λ°©λ¬Έν• κ²½μ° return

        # μ΄ λ• λ‹¤μ qμ—λ” ν„μ¬ λ…Έλ“λ¥Ό μ μ™Έν• λ¦¬μ¤νΈλ΅ μ¬κµ¬μ„±ν•μ—¬ μ¬κ·€ν•¨μ μ”μ²­
        find_max_recursive(next_loc, deepcopy(used), nsheep, nwolf,
                           can_go=[loc for loc in can_go if loc != next_loc and not used[loc]])


def solution(info, edges):

    is_wolf = info  # μ „μ—­λ³€μμ— ν• λ‹Ή
